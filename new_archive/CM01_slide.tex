\section{应用公式}
\begin{frame}{应用公式}
	\begin{enumerate}
		\item $a^{2}-b^{2}=(a+b)(a-b)$.
		\item $a^{3}+b^{3}=(a+b)\left(a^{2}-a b+b^{2}\right)$.
		\item $a^{3}-b^{3}=(a-b)\left(a^{2}+a b+b^{2}\right)$.
		\item $a^{2}+2 a b+b^{2}=(a+b)^{2}$.
		\item $a^{2}-2 a b+b^{2}=(a-b)^{2}$.
		\item $a^{3}+3 a^{2} b+3 a b^{2}+b^{3}=(a+b)^{3}$.
		\item $a^{3}-3 a^{2} b+3 a b^{2}-b^{3}=(a-b)^{3}$.
	\end{enumerate}
\end{frame}
